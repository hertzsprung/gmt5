/*
 * Contains macros for presenting the GMT common options in man page
 * synopsis - yielding consistent presentation from all programs.
 * See gmt_synopsis.h for macros for the C programs.
 *
 * Author:	Paul Wessel
 * Date:	6-APR-2006
 * Version:	4
 *
 */

#ifndef GMT_SYNOPSIS_TXT
#define GMT_SYNOPSIS_TXT

/* Generic option, yields -letter in bold */
#define OPT(letter)	\fB\-letter\fP
/* The GMT standard options */
#define GMT_B_OPT	OPT(B)[\fBp\fP|\fBs\fP]\fIparameters\fP
#define GMT_H_OPT	OPT(H)[\fBi\fP][\fInrec\fP]
#define GMT_I_OPT	OPT(I)\fIxinc\fP[\fIunit\fP][\fB=\fP|\fB+\fP][/\fIyinc\fP[\fIunit\fP][\fB=\fP|\fB+\fP]]
#define GMT_inc_OPT	\fIxinc\fP[\fIunit\fP][\fB=\fP|\fB+\fP][/\fIyinc\fP[\fIunit\fP][\fB=\fP|\fB+\fP]]
#define GMT_Id_OPT	OPT(I)\fIxinc\fP[\fBm\fP|\fBc\fP][/\fIyinc\fP[\fBm\fP|\fBc\fP]]
#define GMT_J_OPT	OPT(J)\fIparameters\fP
#define GMT_Jx_OPT	OPT(Jx)|\fBX\fP\fIparameters\fP
#define GMT_Jz_OPT	OPT(Jz)|\fBZ\fP\fIparameters\fP
#define GMT_K_OPT	OPT(K)
#define GMT_M_OPT	OPT(M)[\fBi\fP|\fBo\fP][\fIflag\fP]
#define GMT_Mo_OPT	OPT(M)[\fIflag\fP]
#define GMT_O_OPT	OPT(O)
#define GMT_P_OPT	OPT(P)
#define GMT_Rgeo_OPT	OPT(R)\fIwest\fP/\fIeast\fP/\fIsouth\fP/\fInorth\fP[\fBr\fP]
#define GMT_Rgeoz_OPT	OPT(R)\fIwest\fP/\fIeast\fP/\fIsouth\fP/\fInorth\fP[/\fIzmin\fP/\fIzmax\fP][\fBr\fP]
#define GMT_Rx_OPT	OPT(R)\fIxmin\fP/\fIxmax\fP/\fIymin\fP/\fIymax\fP[\fBr\fP]
#define GMT_U_OPT	OPT(U)[\fIjust\fP/\fIdx\fP/\fIdy\fP/][\fBc\fP|\fIlabel\fP]
#define GMT_V_OPT	OPT(V)
#define GMT_X_OPT	OPT(X)[\fBa\fP|\fBc\fP|\fBr\fP][\fIx-shift\fP[\fBu\fP]]
#define GMT_Y_OPT	OPT(Y)[\fBa\fP|\fBc\fP|\fBr\fP][\fIy-shift\fP[\fBu\fP]]
#define GMT_c_OPT	OPT(c)\fIcopies\fP
#define GMT_t_OPT	OPT(:)[\fBi\fP|\fBo\fP]
/* Use b, f when applies to both i and o, else use only the bi, bo, fi, fo variants */
#define GMT_b_OPT	OPT(b)[\fBi\fP|\fBo\fP][\fBs\fP|\fBS\fP|\fBd\fP|\fBD\fP[\fIncol\fP]|\fBc\fP[\fIvar1\fP\fB/\fP\fI...\fP]]
#define GMT_bi_OPT	OPT(bi)[\fBs\fP|\fBS\fP|\fBd\fP|\fBD\fP[\fIncol\fP]|\fBc\fP[\fIvar1\fP\fB/\fP\fI...\fP]]
#define GMT_bo_OPT	OPT(bo)[\fBs\fP|\fBS\fP|\fBd\fP|\fBD\fP[\fIncol\fP]|\fBc\fP[\fIvar1\fP\fB/\fP\fI...\fP]]
#define GMT_f_OPT	OPT(f)[\fBi\fP|\fBo\fP]\fIcolinfo\fP
#define GMT_fi_OPT	OPT(f)\fIcolinfo\fP
#define GMT_fo_OPT	OPT(f)\fIcolinfo\fP

/* Options for map rose and scale, used in pscoast and psbasemap */
#define GMT_TROSE	OPT(T)[\fBf\fP|\fBm\fP][\fBx\fP]\fIlon0\fP/\fIlat0\fP/\fIsize\fP[/\fIinfo\fP][:w,e,s,n:][+\fIgint\fP[/\fImint\fP]]
#define GMT_SCALE	OPT(L)[\fBf\fP][\fBx\fP]\fIlon0\fP/\fIlat0\fP[/\fIslon\fP]/\fIslat\fP/\fIlength\fP[\fBm\fP|\fBn\fP|\fBk\fP][:\fIlabel\fP:\fIjust\fP][\fB+p\fP\fIpen\fP][\fB+f\fP\fIfill\fP] ]

/* OPtions for pseudo perspective views */
#define GMT_E_OPT	OPT(E)\fIazim\fP/\fIelev\fP[\fB+g\fP\fIlon\fP/\fIlat\fP[/\fIz\fP]][\fB+c\fP\fIx0\fP/\fIy0\fP]

/* Argument to *contour programs */
#define GMT_CONTG	OPT(G)[\fBd\fP|\fBf\fP|\fBn\fP|\fBl\fP|\fBL\fP|\fBx\fP|\fBX\fP]\fIparams\fP

/* Used in tools that sets grdheader information via a -D option */
#define GMT_GRDEDIT	OPT(D)\fIxname\fP/\fIyname\fP/\fIzname\fP/\fIscale\fP/\fIoffset\fP/\fItitle\fP/\fIremark\fP

/* Optional suffix for grids */
#define GMT_GRD_SUFFIX	\fB=\fP\fIid\fP[\fB/\fP\fIscale\fP\fB/\fP\fIoffset\fP[\fB/\fP\fInan\fP]]

#endif /* GMT_SYNOPSIS_TXT */

#include "gmt_version.h"
